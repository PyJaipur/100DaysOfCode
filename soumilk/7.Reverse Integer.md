class Solution:
    def reverse(self, x):
        s=str(x)
        if x<0:
            s=s[:0:-1]
            s=0-int(s)
            if (s> 2**31 or s<= -2**31):
                return 0
            return s
            
        s=s[::-1]
        s=int(s)
        if (s> 2**31 or s<= -2**31):
            return 0
        return s
